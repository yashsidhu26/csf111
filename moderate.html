💡 How to think about this
1
createCityList: Allocate an array of n char* pointers. Each pointer will later point to one city name string. Don't allocate the strings yet — that's fillCities' job.
2
fillCities: For each city, read the name into a temporary buffer using fgets. Remove the trailing newline ('\n') that fgets adds. Then use strdup() to make a dynamically allocated copy and store its pointer in cities[i].
3
printCities: Simply loop and printf each cities[i].
4
freeCities: First free each individual string (free(cities[i])), then free the outer pointer array (free(cities)). Always free from the inside out.
📌 strdup(s) is shorthand for malloc(strlen(s)+1) + strcpy. It gives you a fresh heap copy of a string. Every strdup must be matched with a free.
c — Solution
char** createCityList(int n) {
    // Allocate an array of n char* pointers
    char** cities = (char**) malloc(n * sizeof(char*));
    return cities;
}

void fillCities(char** cities, int n) {
    char buffer[100];   // temporary storage for each city name
    for (int i = 0; i < n; i++) {
        printf("Enter city %d: ", i + 1);
        fgets(buffer, 100, stdin);

        // fgets includes the '\n' at the end — remove it
        int len = strlen(buffer);
        if (buffer[len - 1] == '\n') {
            buffer[len - 1] = '\0';
        }

        // strdup makes a heap copy of the string
        cities[i] = strdup(buffer);
    }
}

void printCities(char** cities, int n) {
    printf("Cities: ");
    for (int i = 0; i < n; i++) {
        printf("%s", cities[i]);
        if (i < n - 1) printf(", ");
    }
    printf("\n");
}

void freeCities(char** cities, int n) {
    // Step 1: Free each individual string
    for (int i = 0; i < n; i++) {
        free(cities[i]);
    }
    // Step 2: Free the outer pointer array
    free(cities);
}

Example Output
Enter a sentence: Hello World how are you Reversed: you are how World Hello
💡 How to think about this
1
Use strtok(str, " \n") in a loop to split the sentence into individual words. Store each word pointer in an array (e.g., words[50]) and count how many words there are.
2
strtok works by replacing delimiters with '\0' and returning a pointer to each token. The first call takes the string; subsequent calls pass NULL to continue from where it left off.
3
After collecting all words, loop from count - 1 down to 0 and print each word with a space in between.
📌 strtok modifies the original string by inserting null terminators. That's fine here. The key pattern is: first call = strtok(str, delim), every following call = strtok(NULL, delim) until it returns NULL.
c — Solution
void reverseWords(char* str) {
    char* words[50];   // store pointers to each word
    int count = 0;

    // strtok splits str on spaces and newlines
    // First call: pass the actual string
    char* token = strtok(str, " \n");
    while (token != NULL) {
        words[count] = token;   // save the pointer to this word
        count++;
        // Subsequent calls: pass NULL to continue
        token = strtok(NULL, " \n");
    }

    // Now print words from last to first
    printf("Reversed: ");
    for (int i = count - 1; i >= 0; i--) {
        printf("%s", words[i]);
        if (i > 0) printf(" ");
    }
    printf("\n");
}

Example Output

💡 How to think about this
1
addAccount: Allocate a new node, copy the name with strcpy, set the balance, and set next = NULL. If the list is empty, return the new node as the head. Otherwise walk to the last node and attach it.
2
printAccounts: Standard traversal — walk from head to NULL, printing each node's name and balance with printf("%.2f") for two decimal places.
3
freeAccounts: You must save curr->next BEFORE freeing curr, because once you call free(curr) the memory is gone and you can't read curr->next from it safely. Save it as next = curr->next, free curr, then set curr = next.
⚠️ Never access curr->next after calling free(curr). Always save the next pointer FIRST, then free, then advance.
c — Solution
Account* addAccount(Account* head, char* name, float balance) {
    // Create a new account node
    Account* newNode = (Account*) malloc(sizeof(Account));
    strcpy(newNode->name, name);
    newNode->balance = balance;
    newNode->next = NULL;

    // If list is empty, this node is the head
    if (head == NULL) return newNode;

    // Otherwise, walk to the last node and attach
    Account* curr = head;
    while (curr->next != NULL) {
        curr = curr->next;
    }
    curr->next = newNode;
    return head;
}

void printAccounts(Account* head) {
    Account* curr = head;
    while (curr != NULL) {
        printf("Account: %-10s | Balance: %.2f\n",
               curr->name, curr->balance);
        curr = curr->next;
    }
}

void freeAccounts(Account* head) {
    Account* curr = head;
    while (curr != NULL) {
        Account* next = curr->next;   // SAVE next before freeing
        free(curr);                   // free current node
        curr = next;                  // move to the saved next
    }
}

💡 How to think about this

A jagged array is a regular dynamic 2D array EXCEPT each row has a different length. Allocate the outer pointer array normally (nStudents pointers), then for row i, allocate exactly nExamsPerStudent[i] integers.
2

c — Solution
int** createJaggedMarks(int nStudents, int* nExamsPerStudent) {
    // Allocate array of row pointers
    int** marks = (int**) malloc(nStudents * sizeof(int*));

    // Each row gets its own length
    for (int i = 0; i < nStudents; i++) {
        marks[i] = (int*) malloc(nExamsPerStudent[i] * sizeof(int));
    }
    return marks;
}

void fillJaggedMarks(int** marks, int* nExamsPerStudent, int nStudents) {
    for (int i = 0; i < nStudents; i++) {
        printf("Enter %d marks for student %d: ", nExamsPerStudent[i], i + 1);
        for (int j = 0; j < nExamsPerStudent[i]; j++) {
            scanf("%d", &marks[i][j]);
        }
    }
}

void printJaggedMarks(int** marks, int* nExamsPerStudent, int nStudents) {
    for (int i = 0; i < nStudents; i++) {
        printf("Student %d (%d exams): ", i + 1, nExamsPerStudent[i]);
        for (int j = 0; j < nExamsPerStudent[i]; j++) {
            printf("%d ", marks[i][j]);
        }
        printf("\n");
    }
}

void freeJaggedMarks(int** marks, int nStudents) {
    for (int i = 0; i < nStudents; i++) {
        free(marks[i]);   // free each row
    }
    free(marks);          // free the outer array
}

Example Output
1 2 3 4 5 6
💡 How to think about this
1
Allocate a result array of size n1 + n2. Use three index variables: i (position in arr1), j (position in arr2), k (position in merged result).
2
While both arrays still have elements: compare arr1[i] and arr2[j]. Copy the smaller one into merged[k] and advance that array's index plus k.
3
After the loop, one array might still have leftover elements. Copy all remaining elements from whichever array isn't exhausted yet.
📌 This is the "merge" step of merge sort. It's O(n1+n2) time — much faster than sorting the combined array from scratch. No sorting algorithm needed at all!
c — Solution
int* mergeSortedArrays(int* arr1, int n1, int* arr2, int n2) {
    int* merged = (int*) malloc((n1 + n2) * sizeof(int));
    int i = 0;   // index into arr1
    int j = 0;   // index into arr2
    int k = 0;   // index into merged

    // Pick the smaller element from the front of each array
    while (i < n1 && j < n2) {
        if (arr1[i] <= arr2[j]) {
            merged[k] = arr1[i];
            i++;
        } else {
            merged[k] = arr2[j];
            j++;
        }
        k++;
    }

    // Copy leftover elements from arr1 (if any)
    while (i < n1) {
        merged[k] = arr1[i];
        i++;
        k++;
    }

    // Copy leftover elements from arr2 (if any)
    while (j < n2) {
        merged[k] = arr2[j];
        j++;
        k++;
    }

    return merged;
}

💡 How to think about this
1
This is the same pattern as the integer dynamic matrix (Q15 Easy), but with float instead of int. Allocate rows float pointers, then for each row allocate cols floats.
2
sumMatrix: Use a float sum = 0.0. Nested loops, add mat[i][j] to sum each iteration. Return the total after both loops finish.
3

c — Solution
float** createMatrix(int rows, int cols) {
    float** mat = (float**) malloc(rows * sizeof(float*));
    for (int i = 0; i < rows; i++) {
        mat[i] = (float*) malloc(cols * sizeof(float));
    }
    return mat;
}

void fillMatrix(float** mat, int rows, int cols) {
    printf("Enter elements: ");
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            scanf("%f", &mat[i][j]);   // %f for float
        }
    }
}

float sumMatrix(float** mat, int rows, int cols) {
    float sum = 0.0;
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            sum += mat[i][j];   // add every element
        }
    }
    return sum;
}

void freeMatrix(float** mat, int rows) {
    for (int i = 0; i < rows; i++) {
        free(mat[i]);   // free each row's float array
    }
    free(mat);          // free the outer pointer array
}

💡 How to think about this
1
We only need to resize the outer array of pointers (the char**). The actual strings themselves don't move — their pointers just get copied along by realloc automatically.
2
Call realloc(list, newSize * sizeof(char*)). This gives us a bigger array of char pointers. The original "apple" and "orange" pointers are preserved unchanged in the first two slots.
3
Always check the return value of realloc for NULL. Save to a temporary variable first to avoid losing the original pointer if realloc fails.

c — Solution
char** resizeStringList(char** list, int oldSize, int newSize) {
    // Resize the outer pointer array
    // realloc automatically preserves the existing char* pointers
    char** newList = (char**) realloc(list, newSize * sizeof(char*));

    if (newList == NULL) {
        printf("realloc failed!\n");
        // list is still valid here — free it before exiting
        free(list);
        exit(1);
    }

    return newList;
    // The new slots (oldSize to newSize-1) are uninitialized.
    // The caller will fill them in with strdup().
}

💡 How to think about this
1
You can't just do s1 = s2 because s1 and s2 are pointers. You need to dereference: *s1 = *s2 copies the entire struct value.
2
The classic 3-step swap works for entire structs: create a Student temp = *s1, then *s1 = *s2, then *s2 = temp. C copies all fields including the arrays automatically when you assign structs.
3
This is much simpler than swapping field-by-field. Struct assignment in C copies every field including arrays — it's a full value copy.
📌 In C, you CAN assign structs with =. This does a shallow copy of all fields. For arrays inside structs, C copies the array elements. This is different from array-to-array assignment which doesn't work alone.
c — Solution
void swapStudents(Student* s1, Student* s2) {
    // Step 1: Save a copy of *s1 in a temporary struct
    Student temp = *s1;   // copies ALL fields (name, roll, marks[])

    // Step 2: Copy everything from s2 into s1
    *s1 = *s2;

    // Step 3: Copy the saved original s1 into s2
    *s2 = temp;

    // After these 3 lines, s1 and s2 are fully swapped.
    // We do NOT need to work field-by-field; struct assignment does it all.
}

Example Output
Enter string: hello 'h': 1 'e': 1 'l': 2 'o': 1
💡 How to think about this
1
Create an array int freq[256] = {0}. There are 256 possible ASCII characters. Each character's ASCII value is a number (e.g., 'a' = 97). Use the character as an index: freq[(int)str[i]]++.
2
Loop through the string character by character until you hit '\0'. For each character, increment its counter in the freq array.
3
After counting, loop through all 256 possible ASCII values. If freq[i] > 0 and the character is printable (not a control character), print it with its count.
📌 This is a classic "frequency counting" trick. Using a character as an array index works because in C, a char is just a small integer. freq['a'] and freq[97] are the same thing.
c — Solution
void countFrequencies(const char* str) {
    // One slot for every possible ASCII character value (0–255)
    int freq[256] = {0};   // initialize all to zero

    // Walk through the string and tally each character
    int i = 0;
    while (str[i] != '\0') {
        freq[(int) str[i]]++;   // use char value as index
        i++;
    }

    // Print characters that appeared at least once
    // We check printable range (32–126) to skip newlines and control chars
    for (int c = 32; c < 127; c++) {
        if (freq[c] > 0) {
            printf("'%c': %d\n", (char)c, freq[c]);
        }
    }
}

💡 How to think about this
1
Allocate the outer array of n char* pointers. Then loop n times: read each book title with fgets into a buffer, strip the newline, then use strdup(buffer) to store a heap-allocated copy in books[i].
2
strdup calls malloc internally, so each books[i] is a separately allocated string on the heap. This means each one must be individually freed.
3
freeBookList: Free each book title string first, then free the outer pointer array. The order matters — if you free the outer array first, you lose access to the inner pointers and leak memory.

c — Solution
char** createBookList(int n) {
    // Allocate outer array of n char pointers
    char** books = (char**) malloc(n * sizeof(char*));
    char buffer[200];

    for (int i = 0; i < n; i++) {
        printf("Enter book %d: ", i + 1);
        fgets(buffer, 200, stdin);

        // Remove trailing newline added by fgets
        int len = strlen(buffer);
        if (buffer[len - 1] == '\n') buffer[len - 1] = '\0';

        // strdup makes a heap copy of the title string
        books[i] = strdup(buffer);
    }

    return books;
}

void freeBookList(char** books, int n) {
    // Free each title string first
    for (int i = 0; i < n; i++) {
        free(books[i]);
    }
    // Then free the outer pointer array
    free(books);
}

Split Linked List by Even/Odd Values

Story
Given a linked list of integers, split it into two separate lists: one containing all even-valued nodes and another containing all odd-valued nodes, preserving original order.
Requirements
  • Implement void splitList(Node* head, Node** evenHead, Node** oddHead)
  • Use Node** evenHead and Node** oddHead as output parameters (pointer-to-pointer)
  • Do NOT allocate new nodes — reuse existing nodes by relinking
Starter Code
c — Q13.c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

void splitList(Node* head, Node** evenHead, Node** oddHead);

int main() {
    // Build: 1 -> 2 -> 3 -> 4 -> 5 -> NULL (manually for demo)
    // Then call splitList and print both halves
    return 0;
}
Example Output
Original: 1 -> 2 -> 3 -> 4 -> 5 -> NULL Even list: 2 -> 4 -> NULL Odd list: 1 -> 3 -> 5 -> NULL
💡 How to think about this
1
Use two "tail" pointers (evenTail and oddTail) to track the last node added to each sub-list. Building from the tail is how we maintain order.
2
Walk through the original list. For each node, check curr->data % 2 == 0. If even, attach it to the even list's tail; if odd, attach it to the odd list's tail. Advance the appropriate tail pointer.
3
We write to *evenHead and *oddHead (dereferencing pointer-to-pointer) so the caller's variables get updated.
4
After the loop, make sure both sub-lists end with NULL by setting evenTail->next = NULL and oddTail->next = NULL (if they exist).
📌 Node** is a pointer to a pointer. We pass &evenHead from main so the function can update the caller's evenHead variable. Writing *evenHead = ... sets that variable.
c — Solution
void splitList(Node* head, Node** evenHead, Node** oddHead) {
    *evenHead = NULL;
    *oddHead  = NULL;

    Node* evenTail = NULL;   // last node added to even list
    Node* oddTail  = NULL;   // last node added to odd list

    Node* curr = head;
    while (curr != NULL) {
        if (curr->data % 2 == 0) {
            // Attach to even list
            if (*evenHead == NULL) {
                *evenHead = curr;     // first even node becomes head
                evenTail  = curr;
            } else {
                evenTail->next = curr;  // append to tail
                evenTail = curr;
            }
        } else {
            // Attach to odd list
            if (*oddHead == NULL) {
                *oddHead = curr;
                oddTail  = curr;
            } else {
                oddTail->next = curr;
                oddTail = curr;
            }
        }
        curr = curr->next;
    }

    // Terminate both sub-lists
    if (evenTail != NULL) evenTail->next = NULL;
    if (oddTail  != NULL) oddTail->next  = NULL;
}

Sort Students by Roll Number (Bubble Sort)
SortingStructuresModerate
Story
Sort an array of student structs in ascending order by roll number using bubble sort. After sorting, display the sorted list.
Requirements
  • Student struct: int roll, char name[50]
  • Implement void sortStudents(Student* arr, int n) using bubble sort
  • Swap entire Student structs (not just one field)

        {15, "Alice"}
    };
    int n = 3;

        printf("%d %s\n", arr[i].roll, arr[i].name);
    return 0;
}

    // TODO
}
Example Output
10 Bob 15 Alice 20 Charlie
💡 How to think about this
1
Bubble sort: In each pass through the array, compare adjacent elements. If the left one is bigger than the right, swap them. After each full pass, the largest unsorted element "bubbles up" to its correct position.
2
You need n - 1 passes. In each pass, compare elements from index 0 to n - 2 - i (since the last i elements are already sorted). The outer loop counts passes, the inner loop does comparisons.
3
To swap, use a temporary Student variable — Student temp = arr[j]; arr[j] = arr[j+1]; arr[j+1] = temp;. C copies all struct fields including the name array automatically.
📌 Bubble sort is O(n²) — not fast for large data, but perfectly fine for small arrays and exams. It's simple to write and understand. The key is: compare neighbors → swap if out of order → repeat.
c — Solution
void sortStudents(Student* arr, int n) {
    // Bubble sort: n-1 passes through the array
    for (int i = 0; i < n - 1; i++) {
        // Each pass, the largest unsorted item bubbles to the right
        // We don't need to check the last i elements (already sorted)
        for (int j = 0; j < n - 1 - i; j++) {
            // If left roll > right roll, swap the whole structs
            if (arr[j].roll > arr[j + 1].roll) {
                Student temp  = arr[j];
                arr[j]        = arr[j + 1];
                arr[j + 1]    = temp;
                // C copies all fields (including name[]) in struct assignment
            }
        }
    }
}

💡 How to think about this

This is one of the simplest string problems. Use a for or while loop to walk through each character of the string until you hit '\0'.
2
Inside the loop, check if (str[i] == ' '). If true, set str[i] = '-'. This directly modifies the character in memory.

            str[i] = '-';   // replace space with dash directly in memory
        }
        i++;
    }
    // The original string in main() is now modified — no return needed
}

2
For each row i, initialize result[i] = 0. Then loop over all columns j and add mat[i][j] to result[i].

int* sumEachRow(int** mat, int rows, int cols) {
    // Allocate a 1D array to hold one sum per row
    int* result = (int*) malloc(rows * sizeof(int));

        for (int j = 0; j < cols; j++) {
            result[i] += mat[i][j];
        }
    }

💡 How to think about this
1
Edge cases first: If head1 == NULL, return head2 (the second list is the result). If head2 == NULL, return head1 unchanged.
2
Walk curr through the first list until curr->next == NULL. At that point, curr is the tail node of list 1.
3
Set curr->next = head2. This single assignment joins the two lists. Return head1 — the start of the now-combined list.
📌 Concatenation is destructive — the tail of list1 now points into list2. Freeing nodes from both lists later can be done with a single traversal starting from the returned head.
c — Solution
Node* concatenateLists(Node* head1, Node* head2) {
    // Edge case: first list is empty
    if (head1 == NULL) return head2;

    // Edge case: second list is empty
    if (head2 == NULL) return head1;

    // Find the last node of the first list
    Node* curr = head1;
    while (curr->next != NULL) {
        curr = curr->next;   // move to next until we reach the tail
    }

    // Attach the second list to the tail of the first
    curr->next = head2;

    // Return head1 — it's now the head of the full combined list
    return head1;
}

    printf("Modified: %s\n", result);

    // TODO

Example Output
Enter sentence: I love C programming Enter word to replace: C Enter replacement word: Python Modified: I love Python programming
💡 How to think about this
1
Use strstr(str, oldWord) to find the first occurrence of the word in the string. It returns a pointer to where the match starts, or NULL if not found.
2
If not found, return a strdup of the original — an exact copy. The caller expects a heap-allocated string either way.
3
If found: Calculate the position of the match as pos = found - str (pointer subtraction). Allocate a result string big enough for the whole new sentence. Copy the part before the match (pos chars), append newWord, then append everything after the match (found + strlen(oldWord)).
📌 strstr returns a pointer into the original string. Subtracting the start pointer gives the offset (number of chars before the match). strncat and strcat are used to safely build up the result.
c — Solution
char* replaceWord(const char* str, const char* oldWord, const char* newWord) {
    // Find the first occurrence of oldWord in str
    const char* found = strstr(str, oldWord);

    // If oldWord is not in the string, return a copy of the original
    if (found == NULL) {
        return strdup(str);
    }

    // How many characters come BEFORE the match?
    int pos = (int)(found - str);

    // Allocate enough for: part before + newWord + part after + null
    int newLen = strlen(str) - strlen(oldWord) + strlen(newWord) + 1;
    char* result = (char*) malloc(newLen * sizeof(char));
    result[0] = '\0';   // start as empty string

    // 1. Copy the part of str BEFORE the match
    strncat(result, str, pos);

    // 2. Append the new word
    strcat(result, newWord);

    // 3. Append everything AFTER the matched old word
    strcat(result, found + strlen(oldWord));

    return result;   // caller must free() this
}

💡 How to think about this
1
The brute-force approach: for each element in arr1 (outer loop), check every element in arr2 (inner loop). If any element in arr2 matches, print arr1[i] and break out of the inner loop.
2
Breaking out of the inner loop with break after finding the first match prevents printing a common element multiple times even if it appears several times in arr2.
3

📌 Using a break after printing is a simple de-duplication trick. More efficient solutions exist (using hash sets or sorting), but this nested loop approach is the clearest to read and write.
c — Solution
void findCommonElements(int* arr1, int n1, int* arr2, int n2) {
    printf("Common elements: ");

    // For each element in arr1...
    for (int i = 0; i < n1; i++) {
        // ...check if it exists anywhere in arr2
        for (int j = 0; j < n2; j++) {
            if (arr1[i] == arr2[j]) {
                printf("%d ", arr1[i]);
                break;   // stop searching arr2; move to next arr1 element
                // This also prevents printing duplicates if arr2 has repeats
            }
        }
    }
    printf("\n");
}
