Boilerplate, I/O & Compilation

#include <stdio.h>
#include <stdlib.h>   // malloc, free, exit
#include <string.h>   // strlen, strcpy, strcmp etc.
#include <math.h>     // if needed — compile with -lm

int main() {
    // your code
    return 0;
}
c
✓ printf format specifiers
  • %d — int
  • %f — float (default 6 dp)
  • %.2f — float 2 decimal places
  • %lf — double (scanf only)
  • %c — char
  • %s — string
  • %p — pointer address
  • %ld — long int
  • %u — unsigned int
⚠ scanf traps
  • Always use & for primitives: scanf("%d", &x)
  • Do NOT use & for arrays/strings: scanf("%s", str)
  • To read a line with spaces: fgets(str, size, stdin)
  • After scanf int, a \n stays in buffer — use getchar() before next fgets
  • Use %lf for double in scanf, but %f for printf
ℹ compile commands
  • gcc prog.c -o prog
  • gcc prog.c -o prog -lm (for math.h)
  • ./prog to run
  • Multi-file: gcc a.c b.c -o out

⚠ Integer Division Trap

5/2 = 2 (NOT 2.5). To get float result: (float)5/2 or 5/2.0

★ Operator Precedence (High→Low)

() [] -> → ! ~ ++ -- (cast) * & sizeof → * / % → + - → << >> → < <= > >= → == != → & → ^ → | → && → || → ?: → = += -=

ℹ Useful char tricks
  • 'A' = 65, 'a' = 97, '0' = 48
  • Uppercase to lowercase: c + 32 or c - 'A' + 'a'
  • Digit char to int: c - '0'
  • Check digit: c >= '0' && c <= '9'
✓ Pre vs Post Increment

int a=5; int b = a++; → b=5, a=6
int b = ++a; → b=6, a=6
In loops i++ and ++i behave the same.

Type Casting

int a = 7, b = 2;
float result = (float)a / b;     // 3.5 — cast a BEFORE division
float result2 = (float)(a / b); // 3.0 — WRONG, division happens first

Control Flow — if/else, switch, loops

Pattern — FizzBuzz style
for(int i=1; i<=100; i++) {
    if(i%15==0)      printf("FizzBuzz\n");
    else if(i%3==0) printf("Fizz\n");
    else if(i%5==0) printf("Buzz\n");
    else             printf("%d\n", i);
}
c
Pattern — while with sentinel
int n;
scanf("%d", &n);
while(n != -1) {
    // process n
    scanf("%d", &n);
}
c
Switch — remember break!
switch(ch) {
    case 'a': printf("vowel\n"); break;
    case 'e': printf("vowel\n"); break;
    default: printf("consonant\n");
}
// Without break, execution FALLS THROUGH to next case!
c
⚠ Missing break in switch causes fall-through — a very common exam mistake. Always add break unless intentional.
★ Loop Patterns to Know
  • Sum of digits of n: while(n) { sum += n%10; n/=10; }
  • Reverse a number: while(n) { rev = rev*10 + n%10; n/=10; }
  • Count digits: while(n) { count++; n/=10; }
  • GCD (Euclidean): while(b) { t=b; b=a%b; a=t; }
  • Prime check: loop from 2 to √n
✓ do-while vs while

Use do-while when you need the body to execute AT LEAST ONCE — e.g., menu-driven programs:

do {
    printf("Enter (0=quit): ");
    scanf("%d", &ch);
} while(ch != 0);
Pattern — Star/Number Pattern (exam favourite)
// Right triangle of stars, n rows
for(int i=1; i<=n; i++) {
    for(int j=1; j<=i; j++)
        printf("* ");
    printf("\n");
}

// Pyramid
for(int i=1; i<=n; i++) {
    for(int j=1; j<=n-i; j++) printf(" ");
    for(int j=1; j<=2*i-1; j++) printf("*");
    printf("\n");
}
c

Functions

// Always declare prototype BEFORE main if function defined after main
int add(int, int);      // prototype

int main() {
    printf("%d\n", add(3, 4));
    return 0;
}

int add(int a, int b) {
    return a + b;
}
c
⚠ Call by Value vs Reference

C is always call-by-value. To modify a variable inside a function, pass its ADDRESS:

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}
// Call: swap(&x, &y);
✓ Scope Rules
  • Local vars exist only inside their {} block
  • Global vars accessible everywhere — but prefer locals
  • static local var: retains value between calls
  • Function args are local copies — changes don't affect caller (unless pointers)
★ Common Function Patterns
  • Return max of two: return (a>b) ? a : b;
  • Check prime, return 1/0
  • Factorial (iterative)
  • Power function
  • Pass array to function: void fn(int arr[], int n)
Pattern — Stack vs Heap memory
Stack: local variables, function params. Automatically managed. Size fixed at compile time.
Heap: malloc/calloc/realloc. Manually managed — you must free(). Survives function return if pointer passed out.

Arrays, Searching & Sorting

// Declare + initialize
int arr[5] = {10, 20, 30, 40, 50};
int zeros[100] = {0};   // all zeros

// Passing array to function (decays to pointer)
void print(int arr[], int n) {
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
}
// Call: print(arr, 5);
c
Linear Search
int linearSearch(int arr[], int n, int key) {
    for(int i=0; i<n; i++)
        if(arr[i] == key) return i;   // return index
    return -1;   // not found
}
c
Binary Search — SORTED array only
int binarySearch(int arr[], int n, int key) {
    int low=0, high=n-1;
    while(low <= high) {
        int mid = low + (high - low) / 2;  // safer than (low+high)/2
        if(arr[mid] == key) return mid;
        else if(arr[mid] < key) low = mid + 1;
        else high = mid - 1;
    }
    return -1;
}
c
Bubble Sort
void bubbleSort(int arr[], int n) {
    for(int i=0; i<n-1; i++) {
        for(int j=0; j<n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                int t = arr[j]; arr[j] = arr[j+1]; arr[j+1] = t;
            }
        }
    }
}
c
Selection Sort (also asked)
void selectionSort(int arr[], int n) {
    for(int i=0; i<n-1; i++) {
        int minIdx = i;
        for(int j=i+1; j<n; j++)
            if(arr[j] < arr[minIdx]) minIdx = j;
        int t = arr[i]; arr[i] = arr[minIdx]; arr[minIdx] = t;
    }
}
c

Structures

typedef struct {
    int id;
    char name[50];
    float gpa;
} Student;

int main() {
    Student s1 = {1, "Alice", 9.5};
    printf("%s has GPA %.2f\n", s1.name, s1.gpa);

    Student arr[10];           // array of structs
    arr[0].id = 101;
    strcpy(arr[0].name, "Bob");  // DON'T use = for strings inside structs
}
c
⚠ String in struct

You CANNOT do s.name = "Alice". Use strcpy(s.name, "Alice"). Or initialize at declaration time.

✓ Passing struct to function

Pass by value: void fn(Student s) — gets a copy.
Pass by pointer: void fn(Student *s) — access with s->field or (*s).field.

★ Arrow vs Dot

s.name — s is a struct variable
ptr->name — ptr is a pointer to struct
They're equivalent: ptr->name == (*ptr).name

Array of Structures — search by field
void findById(Student arr[], int n, int id) {
    for(int i=0; i<n; i++) {
        if(arr[i].id == id) {
            printf("Found: %s\n", arr[i].name);
            return;
        }
    }
    printf("Not found\n");
}
c

Pointers

int x = 10;
int *p = &x;         // p holds address of x
printf("%d\n", *p);   // dereference: prints 10
*p = 20;             // changes x to 20
printf("%p\n", p);   // prints address
c
★ Pointer Arithmetic
int arr[] = {10,20,30};
int *p = arr;  // points to arr[0]
p++;           // now points to arr[1]
*(p+1)        // = arr[2] = 30
// arr[i] == *(arr+i) always!
⚠ Common Pointer Mistakes
  • Uninitialized pointer: int *p; *p = 5; — CRASH
  • Dangling pointer: using pointer after free()
  • Memory leak: malloc without free
  • Don't forget * when declaring: int *p, q; — q is int, NOT int*
✓ Arrays of Pointers
char *names[] = {"Alice","Bob","Carol"};
printf("%s\n", names[1]); // Bob
Pointer to struct — modify via function
void updateGPA(Student *s, float newGPA) {
    s->gpa = newGPA;       // same as (*s).gpa = newGPA
}
// Call: updateGPA(&s1, 9.8);
c

Dynamic Memory Allocation

#include <stdlib.h>

// malloc — uninitialized memory
int *arr = (int *)malloc(n * sizeof(int));

// calloc — zero-initialized memory
int *arr = (int *)calloc(n, sizeof(int));

// ALWAYS check for NULL
if(arr == NULL) {
    printf("Memory allocation failed!\n");
    exit(1);
}

// Use it like a normal array
for(int i=0; i<n; i++) arr[i] = i*10;

// ALWAYS free when done
free(arr);
arr = NULL;  // good practice to avoid dangling pointer
c
Dynamic array of structs
Student *students = (Student *)malloc(n * sizeof(Student));
if(!students) { perror("malloc"); exit(1); }

students[0].id = 1;
strcpy(students[0].name, "Alice");

free(students);
c
2D dynamic array
int **mat = (int **)malloc(r * sizeof(int *));
for(int i=0; i<r; i++)
    mat[i] = (int *)malloc(c * sizeof(int));

// Use: mat[i][j]

// Free in reverse order
for(int i=0; i<r; i++) free(mat[i]);
free(mat);
c
⚠ For dynamic 2D arrays: free each row first, THEN free the outer pointer. Wrong order = memory corruption.

High-value complete programs

Pattern — Student records with dynamic alloc + sorting
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char name[50];
    float gpa;
} Student;

int cmpGPA(const void *a, const void *b) {
    Student *sa = (Student *)a;
    Student *sb = (Student *)b;
    if(sa->gpa > sb->gpa) return -1;
    if(sa->gpa < sb->gpa) return  1;
    return 0;
}

int main() {
    int n;
    scanf("%d", &n);
    Student *s = (Student *)malloc(n * sizeof(Student));
    if(!s) { printf("malloc failed\n"); return 1; }

    for(int i=0; i<n; i++)
        scanf("%s %f", s[i].name, &s[i].gpa);

    qsort(s, n, sizeof(Student), cmpGPA); // stdlib sort!

    for(int i=0; i<n; i++)
        printf("%s %.2f\n", s[i].name, s[i].gpa);

    free(s);
    return 0;
}
c
Pattern — Reverse array in-place using pointers
void reverse(int *arr, int n) {
    int *left = arr;
    int *right = arr + n - 1;
    while(left < right) {
        int t = *left;
        *left = *right;
        *right = t;
        left++; right--;
    }
}

Killer Tricks, Pitfalls & Exam Strategy

⚠ Top 10 Bugs That Cost Marks
  • Missing return 0; in main
  • Using = instead of == in condition
  • Off-by-one: i <= n vs i < n
  • Forgetting & in scanf for int/float/char
  • Using & in scanf for string/array
  • Integer division when float expected
  • Missing break in switch
  • Not freeing malloc'd memory
  • strcpy instead of = for char arrays
  • Array out of bounds (arr[n] when size is n)
✓ Smart Shortcuts
  • Ternary: max = (a>b) ? a : b;
  • Swap without temp: a^=b; b^=a; a^=b;
  • Check power of 2: n & (n-1) == 0
  • Even/odd: n%2==0 or n&1==0
  • Absolute value: a < 0 ? -a : a or abs(a)
  • Zero-out array: memset(arr, 0, n*sizeof(int))
ℹ 3-hr Exam Time Strategy
  • Read ALL questions first (5 min)
  • Attempt easy questions first, bank marks
  • For each function: write prototype → body → test call
  • Always test with edge cases: n=0, n=1, negative numbers
  • Even partial marks > blank: write struct + function signatures
  • Comment your logic — it signals intent even if code is buggy
★ Harder Topics at Module 10-11 Level
  • Pointer to pointer (int **) for 2D dynamic
  • Function returning malloc'd pointer
  • Modify array size using realloc
  • Struct with pointer member
  • Passing pointer to function that fills an array
